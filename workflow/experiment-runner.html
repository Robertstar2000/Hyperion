# Experiment: Effect of Temperature on Reaction Rates # Generated by Project Hypatia AI import numpy as np import matplotlib.pyplot as plt from scipy.optimize import curve_fit # Define the Arrhenius equation def arrhenius(T, A, Ea, R=8.314): # T: temperature in Kelvin # A: pre-exponential factor # Ea: activation energy in J/mol # R: gas constant in J/(molÂ·K) return A * np.exp(-Ea / (R * T)) # Generate temperature data points (convert from Celsius to Kelvin) temperatures_C = np.array([10, 20, 30, 40, 50]) temperatures_K = temperatures_C + 273.15 # Define true parameters for simulation A_true = 1.2e9 # pre-exponential factor Ea_true = 45200 # activation energy in J/mol # Calculate theoretical reaction rates true_rates = arrhenius(temperatures_K, A_true, Ea_true) # Add some random noise to simulate experimental error np.random.seed(42) # for reproducibility noise = np.random.normal(0, 0.0005, len(temperatures_K)) measured_rates = true_rates + noise # Fit the Arrhenius equation to the measured data popt, pcov = curve_fit(arrhenius, temperatures_K, measured_rates, p0=[1e9, 40000]) A_fit, Ea_fit = popt # Calculate fitted rates fitted_rates = arrhenius(temperatures_K, A_fit, Ea_fit) # Calculate R-squared value ss_tot = np.sum((measured_rates - np.mean(measured_rates))**2) ss_res = np.sum((measured_rates - fitted_rates)**2) r_squared = 1 - (ss_res / ss_tot) # Save results to a file results = np.column_stack((temperatures_C, measured_rates, fitted_rates)) np.savetxt('experiment_results.csv', results, delimiter=',', header='Temperature (C),Measured Rate,Fitted Rate') # Print results print('Experiment Results:') print(f'Fitted pre-exponential factor (A): {A_fit:.3e}') print(f'Fitted activation energy (Ea): {Ea_fit:.1f} J/mol') print(f'R-squared value: {r_squared:.4f}')
